# Resources check
oxr_mdlc_councilor_editor_check_trait_resources_cost_common = {
    optimize_memory
    custom_tooltip_fail = {
        text = oxr_mdlc_councilor_editor_trait_cost.fail_common
        has_country_resource = { type = energy amount >= @oxr_mdlc_councilor_editor_common_trait_cost }
